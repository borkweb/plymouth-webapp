encoded: Boolean(default: true)
Indicates whether the Editor should submit encoded HTML tags.

Example

$("#editor").kendoEditor({
     encoded: false
 });
stylesheets: Array
Allows custom stylesheets to be included within the editing area.

Example

$("#editor").kendoEditor({
     stylesheets: [
         "common-styles.css",
         "green-theme.css",
     ]
 });
tools: Array
A collection of tools that should render a button, combobox, etc, to interact with the Editor

Example

$("#editor").kendoEditor({
     tools: [
         "bold",
         "italic",
         "underline",
         "foreColor",
         "insertUnorderedList",
         "insertOrderedList",
         "createLink",
         "unlink",
         "insertImage"
     ]
 });
